async def chat_with_companion(user, message, history, db):
  # Build dynamic system prompt
  pending_tasks = await get_pending_tasks_summary(user.id, db)
  recent_mood = await get_latest_mood(user.id, db)
  system = f'''
    You are a warm, supportive Smart Study Companion AI for {user.name}.
    Learning style: {user.learning_style}.
    Pending tasks: {pending_tasks}.
    Current stress level: {recent_mood.stress_level if recent_mood else 'unknown'}/10.
    Keep responses concise (max 150 words). Be encouraging and practical.
    If stress > 7, gently suggest a break or breathing exercise.
  '''
  messages = [{'role': m.role, 'content': m.content} for m in history[-10:]]
  messages.append({'role': 'user', 'content': message})
  response = client.messages.create(
    model='claude-sonnet-4-5',
    max_tokens=500,
    system=system,
    messages=messages
  )
  return response.content[0].text

async def summarize_notes(text: str) -> dict:
  prompt = f'Summarize these study notes into 5 key bullet points:\n\n{text}'
  response = client.messages.create(
    model='claude-sonnet-4-5', max_tokens=400,
    messages=[{'role': 'user', 'content': prompt}]
  )
  return {'summary': response.content[0].text}

── SCHEDULER SERVICE (app/services/scheduler_service.py) ─────────────
async def generate_schedule(user_id, start_date, days, db):
  tasks = await get_pending_tasks(user_id, db)  # ordered by priority_score desc
  user = await get_user(user_id, db)
  peak_hours = user.peak_hours or [9, 14, 19]
  schedules = []
  for day_offset in range(days):
    date = start_date + timedelta(days=day_offset)
    slots = ml.study_scheduler.allocate_day(
      tasks=tasks, date=date, peak_hours=peak_hours,
      max_study_hours=8, pomodoro_work=25, pomodoro_break=5
    )
    schedule = Schedule(user_id=user_id, date=date,
                        time_slots=slots, generated_by_ai=True)
    db.add(schedule)
    schedules.append(schedule)
  await db.commit()
  return schedules

── ML: STUDY SCHEDULER (app/ml/study_scheduler.py) ──────────────────
def allocate_day(tasks, date, peak_hours, max_study_hours, pomodoro_work, pomodoro_break):
  available_start = datetime.combine(date, time(8, 0))  # 8 AM
  available_end = datetime.combine(date, time(22, 0))   # 10 PM
  current_time = available_start
  slots = []
  total_study_mins = 0
  max_study_mins = max_study_hours * 60
  for task in tasks:  # already sorted by priority
    if total_study_mins >= max_study_mins: break
    if current_time >= available_end: break
    # Prefer peak hours for high-difficulty tasks
    if task.difficulty >= 4 and current_time.hour not in peak_hours:
      continue  # defer to later iteration
    duration = min(task.estimated_mins, pomodoro_work * 4)  # max 100 min block
    end_time = current_time + timedelta(minutes=duration)
    if end_time > available_end: break
    slots.append({'start': current_time.isoformat(), 'end': end_time.isoformat(),
                  'task_id': str(task.id), 'task_title': task.title,
                  'slot_type': 'study'})
    current_time = end_time + timedelta(minutes=pomodoro_break)
    # Add break slot
    slots.append({'start': end_time.isoformat(),
                  'end': current_time.isoformat(),
                  'task_id': None, 'task_title': 'Break', 'slot_type': 'break'})
    total_study_mins += duration
  return slots

── ML: STRESS DETECTOR (app/ml/stress_detector.py) ──────────────────
def analyze_stress_pattern(mood_entries: list) -> dict:
  if not mood_entries: return {'alert_triggered': False, 'trend': 'stable'}
  scores = [m.stress_level for m in mood_entries]
  rolling_3 = sum(scores[:3]) / min(3, len(scores))
  # Linear trend via least squares
  x = list(range(len(scores)))
  n = len(x)
  slope = (n*sum(i*s for i,s in zip(x,scores)) - sum(x)*sum(scores))
          / (n*sum(i**2 for i in x) - sum(x)**2 + 1e-9)
  trend = 'increasing' if slope > 0.3 else 'decreasing' if slope < -0.3 else 'stable'
  alert = rolling_3 > 7 and trend in ('increasing', 'stable')
  recs = []
  if alert:
    recs = ['Take a 15-min walk between study sessions',
            'Try 4-7-8 breathing: inhale 4s, hold 7s, exhale 8s',
            'Break your next task into 10-min micro-goals',
            'Talk to a friend or counselor — you are not alone']
  return {'alert_triggered': alert, 'current_avg': rolling_3,
          'trend': trend, 'recommendations': recs}
